SQL QueryGROUP BY · SUBQUERY · LIMIT 10
SELECT
    ph.patient_name,
    COUNT(ph.id)  AS total_visits,
    (
        SELECT SUM(c.medicine_cost + c.doctor_fee
               + COALESCE(c.admission_cost, 0))
        FROM   costs c
        WHERE  c.patient_name = ph.patient_name
    ) AS total_spent
FROM     patienthistory ph
GROUP BY ph.patient_name
ORDER BY total_visits DESC
LIMIT 10;
